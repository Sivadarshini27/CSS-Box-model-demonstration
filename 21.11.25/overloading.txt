package nov20;
public class OverLoading {

    // Method Overloading
    int add(int a, int b) {
        return a + b;
    }

    int add(int a, int b, int c) {
        return a + b + c;
    }

    public static void main(String[] args) {
        OverLoading obj = new OverLoading();

        System.out.println("Sum (2 values): " + obj.add(10, 20));
        System.out.println("Sum (3 values): " + obj.add(10, 20, 30));
    }
}

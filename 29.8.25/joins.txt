sid int,

create table enrollment(
 
sid int,
  
cid int

);

(2,105),

(3,106);



select * from enrollment;



select student.sid, student.sname, list.cid, list.cname, list.dept

from student 
inner join enrollment on student.sid = enrollment.sid
inner 
join list on enrollment.cid = list.cid;



select *from list left outer join student on (cid='sid.cid');

select *from list right outer join student on (cid='sid.cid');



select * 
from enrollment
natural join student;

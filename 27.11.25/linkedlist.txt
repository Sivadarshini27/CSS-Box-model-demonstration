package nov27;
import java.util.*;  
public class LinkedListts{  
public static void main(String args[]){  
LinkedList<String> al=new LinkedList<String>();  
al.add("Magi");  
al.add("Vijay");  
al.add("Surya");  
al.add("Arun");  
Iterator<String> itr=al.iterator();  
while(itr.hasNext()){  
System.out.println(itr.next());  
}  
}  
}
